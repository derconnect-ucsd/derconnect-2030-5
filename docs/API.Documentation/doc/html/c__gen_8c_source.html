346 void compute_flags (List *sorted, SchemaType *head) {
347  SchemaType *t, *u; List *l, *m; Flag *f; int global_bits;
348  List *groups = NULL, *g;
349  // collect flags
350  foreach (l, sorted) { t = l->data;
351  if (t->kind == ComplexType
352  && !t->restriction) {
353  attr_flags (t, head);
354  part_flags (t, t->part, head);
355  }
356  }
357  // distribute flags to types
358  foreach (f, flags) {
359  int n = list_length (f->types);
360  if (n == 1) {
361  t = f->types->data;
362  t->flags = list_insert (t->flags, f);
363  } else
364  groups = add_flag_group (groups, f);
365  }
366  groups = pre_flag_groups (groups, head);
367  global_bits = flag_bits (0, global_flags);
368  foreach (l, sorted) { t = l->data;
369  if (t->kind != ComplexType) continue;
370  if (t->base && (u = find_by_name (head, t->base)))
371  t->base_bits = u->base_bits + u->bits;
372  else
373  foreach (m, global_flags) {
374  Flag *f = m->data;
375  if (find_by_data (f->types, t)) {
376  t->base_bits = global_bits; break;
377  }
378  }
379  t->bits = flag_bits (t->base_bits, t->pre);
380  t->flags = list_reverse (t->flags);
381  t->bits += flag_bits (t->base_bits+t->bits, t->flags);
382  if (t->flags || t->base_bits)
383  set_has_flags (t, head);
384  }
385  post_flag_groups (groups, head);
386 }

458 /* The number of EXI productions in a given parsing context is determined by
459  the number of possible attributes or elements that can occur after the
460  minimum attribute or element count is met for the current table entry.
461  If the minimal count is equal to the maximal count for the current table
462  entry or the minimal count is not met, then the number of productions is 1.
463 */
464 int productions (TableEntry *entries) {
465  int n = 0, first = 1; TableEntry *e;
466  foreach (e, entries) {
467  SchemaElement *se = e->se; n++;
468  if (first && (se->min < se->max || se->unbounded)) first = 0;
469  else if (se->min) break;
470  } if (!e) n++; // reached the end of a sequence
471  return n;
472 }
