136 int tls_session (void *conn) {
137  Connection *c = conn; int err;
138  if (tcp_status (c) == Connected) {
139  switch (c->tls_state) {
140  case TLS_NEGOTIATE:
141  if (ssl_handshake (c->tls)) {
142  c->tls_state = TLS_SESSION; return SESSION_NEW;
143  } else if (!ssl_pending ()) {
144  c->tls_state = TLS_SHUTDOWN;
145  print_ssl_error ("tls_session");
146  } else break;
147  case TLS_SHUTDOWN:
148  if (ssl_close (c->tls)) {
149  c->tls_state = TLS_NONE;
150  ssl_free (c->tls); c->tls = NULL;
151  tcp_close (c);
152  } break;
153  case TLS_SESSION: return SESSION_CONNECTED;
154  }
155  } return SESSION_NONE;
156 }
