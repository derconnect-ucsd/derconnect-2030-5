63 void edev_gen (int i, char *line) {
64  char href[64]; uint64_t sfdi;
65  SE_EndDevice_t *edev;
66  SE_Registration_t *reg;
67  Device *d; Protected *r;
68  if (!number64 (&sfdi, strtok (line, " \t"))) {
69  printf ("devices.txt, line %d: error parsing SFDI\n", i); exit (0);
70  }
71  if (find_device (sfdi)) return;
72  edev = get_edev (sfdi);
73  // EndDevice
74  // device id is the same as the edev list position
75  d = add_device (get_id (edev->href), edev->href, sfdi);
76  // Registration
77  if (se_exists (edev, RegistrationLink)) {
78  reg = resource_data (find_resource (edev->RegistrationLink.href));
79  } else {
80  reg = type_alloc (SE_Registration_t);
81  sprintf (href, "%s/rg", edev->href);
82  reg->href = strdup (href);
83  add_resource (reg, SE_Registration, sfdi);
84  edev->RegistrationLink.href = strdup (href);
85  }
86  if (!number64 (&reg->dateTimeRegistered, strtok (NULL, " \t"))) {
87  printf ("devices.txt, line %d: error parsing timestamp\n", i); exit (0);
88  }
89  reg->pIN = atoi (strtok (NULL, " \t"));
90  // PowerStatus
91  sprintf (href, "%s/ps", edev->href);
92  d->ps = add_empty (SE_PowerStatus, href, sfdi);
93  edev->PowerStatusLink.href = strdup (href);
94  edev->_flags |= SE_PowerStatusLink_exists
95  | SE_RegistrationLink_exists;
96 }
