8 char *dns_name (char *name, char *data) {

10  int n, name_length = 0, prev_length = -1;
11  while (!truncated (data+1)) {
12  if (*data == '\0') {
13  *name = '\0';
14  if (ptr) return ptr+2;
15  return data+1;
16  }

21  if (truncated (data+n)) return NULL;

25  case 0xc0: // pointer

30  if (prev_length == name_length) return NULL; // pointer loop with no data
