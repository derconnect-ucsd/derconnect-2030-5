102 // return pointer to rr fields, name, and length of rdata
103 char *parse_rr (char *name, int *length, char *data) { char *next;
104  if ((data = dns_name (name, data)) && !truncated (data+10)) {

142 // validate DNS-SD TXT record
143 int txt_valid (char *data, int length) {
144  if (length < 2) return 0; // empty TXT record
145  while (length > 0) { int n = *data;
146  if (n) { n++; length -= n; data += n; } else return 0;
147  } return length == 0;
148 }

150 // extract key=value pair from TXT record
151 char *txt_pair (char *key, char **value, char *data) {
152  int n = *data++, m; char *next = data+n; *value = "";
153  if (n) { char *eq = strchr (data, '=');
154  if (eq && (m = eq - data) < n) {
155  n -= m+1; *value = key+m+1;
156  memcpy (*value, eq+1, n);
157  (*value)[n] = '\0'; n = m;
158  } memcpy (key, data, n); key[n] = '\0';
159  return next;
160  } return NULL;
161 }

165  if (data) { char *value;
166  while (data = txt_pair (buffer, &value, data))

239 // process SRV resource record (RFC 2782)
240 void *dns_srv (Service *s, char *data) {
241  char target[256]; int length;
242  s->ttl = rr_ttl (data);
243  s->port = UNPACK16 (data+10+4);
244  ok (dns_name (target, data+10+6));
245  if (!s->host) s->host = get_host (target);
246  if (data = dns_find (target, &length, AAAA_RECORD))
247  host_aaaa (s->host, data, length);
248  else if (data = dns_find (target, &length, A_RECORD))
249  host_a (s->host, data, length);
250  s->srv_found = 1; return data;
251 }

254 void *dns_ptr (char *name, char *data) {
255  Question *q; Service *s; char instance[256]; int length;
256  ok ((q = find_question (name))
257  && (data = dns_name (instance, data+10)));
258  s = get_service (instance); s->query = q->name;
259  if (data = dns_find (instance, &length, SRV_RECORD))

265 // process answer section
266 void *process_answers (char *data, int count) {
267  Service *s; Host *h; int length; char name[256];
268  while (count--) {
269  ok (data = parse_rr (name, &length, data));
270  switch (rr_type (data)) {
271  case PTR_RECORD: dns_ptr (name, data); break;
272  case SRV_RECORD: s = find_service (name);
273  if (s) dns_srv (s, data); break;
274  case TXT_RECORD: s = find_service (name);
275  if (s) dns_txt (s, data, length); break;
276  case A_RECORD: h = find_host (name);
277  if (h) host_a (h, data, length); break;
278  case AAAA_RECORD: h = find_host (name);
279  if (h) host_aaaa (h, data, length); break;
280  }
281  data = rr_next (data, length);
282  }
283 }

285 // process a DNS-SD packet
286 void *dnssd_packet (char *data, int length) { DnsHeader header;
287  dns_start = data; dns_end = data + length; // set the boundry
288  // flags should indicate a standard response with no errors
289  ok ((data = dns_header (&header, data))
290  && (header.flags & 0xf80f) == 0x8000
291  && (data = skip_questions (data, header.qdcount)));
292  ar_section = skip_section (data, header.ancount+header.nscount);
293  process_answers (data, header.ancount);
294 }

296 // add a question to a DNS query packet
297 char *dnssd_question (char *dest, char *name, int type, int unicast) {
298  static char *start, *names[14]; static int count;
299  if (name) { count++; get_question (name);
300  dest = encode_name (dest, name, start, names);
301  PACK16 (dest, type);
302  PACK16 (dest+2, (unicast << 15) | INTERNET_CLASS);
303  PACK16 (start+4, count); // update number of questions
304  return dest+4;
305  }
306  memset (names, 0, sizeof (char *) * 14);
307  start = dest; count = 0; return NULL;
308 }

317 // followup query to request any missing records
318 void dnssd_followup (UdpPort *p) {
319  char packet[1500]; int followup = 0;
320  Service *s = dns_service;
321  char *data = dnssd_query (packet);
322  while (s) { int complete = 1;
323  if (!s->txt_found || !s->srv_found) {
324  data = dnssd_question (data, s->name, ANY_RECORD, 0); complete = 0;
325  }
326  if (!s->host->addr.length) {
327  data = dnssd_question (data, s->host->name, ANY_RECORD, 0); complete = 0;
328  }
329  followup |= !complete; s->complete = complete; s = s->next;
330  }
331  if (followup) udp_send (p, packet, data - packet, &multicast);
332 }
