151 int record_present (Responder *r, char *name, int bit, char *path) {
152  char *packet = encode_name (r->packet, name);
153  int ptr = UNPACK16 (r->packet);
154  if ((ptr & 0xc000) == 0xc000) { int j = 0;
155  while (j < i) {
156  if (r->name[j] == ptr) { bit <<= (j << 1);
157  if (r->ar & bit) return 1;
158  goto update_record;
159  } j++;
160  }
161  }
162  r->name[r->i] = 0xc000 | (r->packet-r->start);
163  r->path[r->i] = path; bit <<= (r->i << 1); r->i++;
164  update_record:
165  r->ar |= bit; r->packet = packet; return 0;
166 }
