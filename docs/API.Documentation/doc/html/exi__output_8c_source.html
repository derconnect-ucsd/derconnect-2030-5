21 void output_byte (Output *o, uint8_t b) {
22  if (o->bit) {
23  *o->ptr++ |= b >> o->bit;
24  *o->ptr = b << (8-o->bit);
25  } else *o->ptr++ = b;

36  while (n > 1 && *b == 0) n--, b++;

42 void output_bit (Output *o, char bit) {

44  *o->ptr++ |= bit; o->bit = 0;
45  } else *o->ptr |= bit << (7 - o->bit++);

48 void output_bits (Output *o, uint32_t bits, int n) {
49  char *ptr = o->ptr;
50  int m = (o->bit + n) & 7;
51  int i = (o->bit + n) >> 3;
52  o->bit = m; o->ptr += i;
53  if (m) { ptr[i] |= bits << (8-m); bits >>= m; }
54  while (i) { ptr[--i] |= bits; bits >>= 8; }
55 }

70 int exi_output_string (Output *o, const SchemaElement *se, char *s) {
71  const char *name = se_name (se, o->schema);
72  StringTable *t = find_table (o->local, name); int i, part, bits;
73  if (t) {
74  i = find_string (t, s);
75  if (i >= 0) { part = 0; goto compact_id; }
76  } else t = o->local = new_string_table (o->local, name, 8);
77  o->local = add_string (o->local, t, s);
78  i = find_string (o->global, s);
79  if (i >= 0) { part = 1; t = o->global; goto compact_id; }
80  o->global = add_string (o->global, o->global, s);
81  exi_output_literal (o, s);
82  return 1;
83  compact_id:
84  output_uint (o, part);
85  bits = bit_count (t->index-1);
86  if (bits) output_bits (o, i, bits);

90 int exi_output_value (Output *o, void *value) {
91  int type = o->se->xs_type;
92  int n = type >> 4;
93  if (o->end - o->ptr < 10) return 0;
94  switch (type & 0xf) {
95  case XS_STRING: if (n) return exi_output_string (o, o->se, value);
96  case XS_ANY_URI: return exi_output_string (o, o->se, *(char **)value);
97  case XS_BOOLEAN: output_bit (o, ((*(uint32_t *)value) >> o->flag) & 1); break;
98  case XS_HEX_BINARY: return output_binary (o, value, n);
99  case XS_LONG: output_integer (o, *(int64_t *)value); break;
100  case XS_INT: output_integer (o, *(int32_t *)value); break;
101  case XS_SHORT: output_integer (o, *(int16_t *)value); break;
102  case XS_BYTE: output_bits (o, (*(int8_t *)value)+128, 8); break;
103  case XS_ULONG: output_uint (o, *(uint64_t *)value); break;
104  case XS_UINT: output_uint (o, *(uint32_t *)value); break;
105  case XS_USHORT: output_uint (o, *(uint16_t *)value); break;
106  case XS_UBYTE: output_bits (o, *(uint8_t *)value, 8); break;
107  } return 1;
108 }

110 int exi_output_simple (Output *o, void *value) {
111  int type = o->se->xs_type;
112  int n = type >> 4; char *s;
113  switch (type & 0xf) {
114  case XS_STRING: if (n) { s = value; break; }
115  case XS_ANY_URI: s = *(char **)value; break;
116  default: output_ch:
117  output_bit (o, 0); // CH
118  return exi_output_value (o, value);
119  }
120  if (*s == '\0') // don't encode empty strings
121  output_bits (o, 0x4, 3); // EE
122  else goto output_ch;
123  return 1;
124 }
