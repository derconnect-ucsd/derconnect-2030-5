21 /* The EXI bit-stream is encoded within a byte-stream, if the number of bytes
22  needed from the byte stream is not met, set the truncated bit and return 0.
23 */
24 #define need(p, n) if ((p->end - p->ptr) < (n)) return p->truncated = 1, 0

26 uint8_t parse_byte (Parser *p) {
27  need (p, p->bit? 2 : 1);
28  if (p->bit) {
29  uint8_t b = (*p->ptr << p->bit) | (*(p->ptr+1) >> (8-p->bit));
30  p->ptr++; return b;
31  } return *p->ptr++;
32 }

34 // parse unsigned integers up to 70 (7x10) bits
35 uint64_t parse_uint (Parser *p) {
36  uint64_t x = 0; int n = 0; uint8_t b;
37  do { b = parse_byte (p);
38  x |= (b & 0x7f) << n; n += 7;
39  } while (!p->truncated && b & 0x80 && n < 70);
40  return x;
41 }

43 /* parse binary data encoded as an unsigned integer and the same number of
44  bytes, the length must be less than or equal to the predefined length by
45  the schema */
46 int parse_binary (Parser *p, uint8_t *b, uint64_t n) {
47  uint64_t m = parse_uint (p);
48  if (p->truncated) return 0;
49  if (m > n) { p->state = PARSE_INVALID; return 0; }
50  need (p, p->bit? m+1 : m);
51  b += n-m;
52  while (m--) *b++ = parse_byte (p); return 1;
53 }

55 int parse_bit (Parser *p) { need (p, 1);
56  int bit = (*p->ptr >> (7-p->bit++)) & 1;
57  if (p->bit & 8) { p->ptr++; p->bit = 0; }
58  return bit;
59 }

61 // parse n bits from the bit stream
62 uint32_t parse_bits (Parser *p, int n) {
63  uint32_t bits = *p->ptr;
64  int bit = (p->bit + n) & 7; // final bit offset
65  int bytes = (p->bit + n) >> 3; // number of bytes to fetch
66  need (p, bit? bytes+1 : bytes); p->bit = bit;
67  while (bytes--) bits = (bits << 8) | *(++p->ptr);
68  return (bits >> (8-bit)) & ~(-1 << n);
69 }

71 // parse a signed integer (bounded range has greater than 4096 values)
72 int64_t parse_integer (Parser *p) {
73  uint8_t *ptr = p->ptr; int bit = p->bit;
74  int sign = parse_bit (p);
75  int64_t x = parse_uint (p);
76  if (p->truncated) { p->ptr = ptr; p->bit = bit; }
77  return sign? -x : x;
78 }

88  if (p->truncated) return 0;

92  } else { *(char **)value = strdup (s); return 1; }

97 // length of a utf-8 string (encoded as an exi string)
98 int exi_utf8_length (Parser *p, int n) {
99  uint8_t *ptr = p->ptr; int length = 0;
100  while (n) { int m = 1, b;
101  /* UTF-8 code points need at most 21 bits (3 bytes in EXI encoding) and
102  are encoded using 1 to 4 bytes in UTF-8. */
103  while ((b = parse_byte (p)) & 0x80)
104  if (++m > 3) { p->state = PARSE_INVALID; return -1; }
105  if (p->truncated) return -1;
106  // adjust encoded length based upon bits needed
107  switch (m) {
108  case 2: if (b & 0x70) m++; break; // more than 11 bits needed?
109  case 3: if (b & 0x7c) m++; break; // more than 16 bits needed?
110  } length += m; n--;
111  } p->ptr = ptr; return length;
112 }

114 // parse an EXI string, either a compact identifier or string literal
115 int exi_parse_string (Parser *p, void *value, int n) {
116  const SchemaElement *se = p->se; StringTable *t;
117  const char *name = se_name (se, p->schema);
118  uint8_t *ptr = p->ptr; char *s;
119  int m = parse_uint (p);
120  if (p->truncated) return 0;
121  switch (m) {
122  case 0: // local value lookup
123  return parse_compact_id (p, find_table (p->local, name), value, n);
124  case 1: // global value lookup
125  return parse_compact_id (p, p->global, value, n);
126  default: // literal value encoding
127  m = exi_utf8_length (p, m-2);
128  if (m < 0) { p->ptr = ptr; return 0; }
129  if (n) { // string is stored in a fixed container
130  if (m >= n) { p->state = PARSE_INVALID; return 0; }
131  s = value;
132  } else { *(char **)value = s = malloc (m+1); }
133  parse_literal (p, s, m);
134  t = find_table (p->local, name);
135  if (!t) t = p->local = new_string_table (p->local, name, 8);
136  p->local = add_string (p->local, t, s);
137  p->global = add_string (p->global, p->global, s);
138  } return 1;
139 }

141 int exi_parse_value (Parser *p, void *value) {
142  int type = p->se->xs_type, n = type >> 4;
143  switch (type & 0xf) {
144  case XS_STRING: return exi_parse_string (p, value, n);
145  case XS_BOOLEAN: if (parse_bit (p)) *(uint32_t *)value |= 1 << p->flag; break;
146  case XS_HEX_BINARY: return parse_binary (p, value, n);
147  case XS_ANY_URI: return exi_parse_string (p, value, 0);
148  case XS_LONG: *(int64_t *)value = parse_integer (p); break;
149  case XS_INT: *(int32_t *)value = parse_integer (p); break;
150  case XS_SHORT: *(int16_t *)value = parse_integer (p); break;
151  case XS_BYTE: *(int8_t *)value = parse_bits (p, 8) - 128; break;
152  case XS_ULONG: *(uint64_t *)value = parse_uint (p); break;
153  case XS_UINT: *(uint32_t *)value = parse_uint (p); break;
154  case XS_USHORT: *(uint16_t *)value = parse_uint (p); break;
155  case XS_UBYTE: *(uint8_t *)value = parse_bits (p, 8); break;
156  } return !p->truncated;
157 }

159 /* IEEE 2030.5-2017 EXI options document in XML format
160  <header xmlns="http://www.w3.org/2009/exi">
161  <common><schemaId>S1</schemaId></common>
162  </header> */
163 /* IEEE 2030.5-2017 EXI options document in EXI format (bitstream)
164  0 (header) | 01 (common) | 10 (schemaID) | 0 (CH)
165  | 0x04 (literal string length=2) | 0x53 0x31 ("S1") | 1 (EE) */

167 /* IEEE 2030.5 uses a fixed set of options, so we only check for the header
168  and the options document as defined above. */
169 int exi_parse_header (Parser *p) { int c;
170  if (memcmp (p->ptr, "$EXI", 4) == 0) p->ptr += 4;
171  need (p, 5); // need 5 bytes
172  c = *p->ptr++; // EXI header is one byte in this case
173  // 10 (distinguishing bits) | 1 (options present) | 00000 (version = 1)
174  if (c == 0xa0) {
175  char schemaId[64];
176  uint32_t x = parse_bits (p, 6);
177  int n = parse_uint (p);
178  if (n && n < 64) { n -= 2;
179  parse_literal (p, schemaId, n);
180  // verify options document
181  if (x == 0xc && streq (schemaId, p->schema->schemaId))
182  return parse_bit (p); // EE
183  }
184  } p->state = PARSE_INVALID; return 0;
185 }

213 int exi_parse_next (Parser *p) {
214  const SchemaElement *se = p->se;
215  if (se->n) {
216  if (p->need_token) {
217  int n = se->min? 1 : se->n;
