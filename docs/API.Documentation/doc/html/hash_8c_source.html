91 // SparseGroup contains up to 58 elements
92 // count and occupancy are marked by bits

94  void **slot; // elements
95  uint64_t bits; // count (6) / bitmap (58)
96 } SparseGroup;

98 // count the bits up to bit position i
99 // https://graphics.stanford.edu/~seander/bithacks.html
100 int bit_rank (uint64_t bits, int i) {
101  if (i == 0) return 0;
102  // Shift out bits after given position.
103  uint64_t r = bits << (64 - i);
104  // Count set bits in parallel.
105  // r = (r & 0x5555...) + ((r >> 1) & 0x5555...);
106  r = r - ((r >> 1) & ~0UL/3);
107  // r = (r & 0x3333...) + ((r >> 2) & 0x3333...);
108  r = (r & ~0UL/5) + ((r >> 2) & ~0UL/5);
109  // r = (r & 0x0f0f...) + ((r >> 4) & 0x0f0f...);
110  r = (r + (r >> 4)) & ~0UL/17;
111  // r = r % 255;
112  r = (r * (~0UL/255)) >> 56;
113  return (int)r;
114 }

116 // insert data into sparesgroup g at position i
117 void sg_insert (SparseGroup *g, int i, void *data) {
118  int s, count, size;
119  uint64_t bit = 1ull << i;
120  s = bit_rank (g->bits, i);
121  count = g->bits >> 58; count++;
122  // printf ("sg_insert %x %d %d %d\n", g, i, s, count);
123  size = count * sizeof (void *);
124  if (g->slot) {
125  g->slot = realloc (g->slot, size);
126  memmove (g->slot+s+1, g->slot+s, (count - (s+1)) * sizeof (void *));
127  } else g->slot = calloc (1, size);
128  g->bits += 1ull << 58; // add one to the count
129  g->bits |= bit; // mark the element as occupied
130  g->slot[s] = data;
131 }

145 void *hash_next (HashTable *ht) {
146  int count = ht->g->bits >> 58;
147  if (++ht->i == count) {
148  next_group:
149  if (ht->g == ht->last) return NULL;
150  ht->g++; count = ht->g->bits >> 58; ht->i = 0;
151  }
152  while (ht->i < count) {
153  void *data = ht->g->slot[ht->i];
154  if (data) return data; ht->i++;
155  } goto next_group;
156 }

170 // return pointer to hash entry with given key or NULL if non-existent
171 void **hash_find (HashTable *ht, void *key) {
172  int mask = ht->size - 1, probes = 0;
173  int index = ht->hash (key) & mask;
174  ht->g = NULL;
175  while (1) {
176  SparseGroup *g = ht->table + index / 58;
177  int i = index % 58; // group index
178  if (!g->slot || sg_empty (g, i)) {
179  if (!ht->g) hash_mark (ht, g, i);
180  return NULL;
181  } else {
182  void **e = sg_element (g, i);
183  if (*e) {
184  if (ht->compare (key, ht->get_key (*e)) == 0)
185  return e;
186  } else if (!ht->g) // deleted element
187  hash_mark (ht, g, i); // mark the location
188  }
189  index = (index + ++probes) & mask;
190  }
191 }

274 // Thomas Wang - 64bit hash function
275 // https://gist.github.com/badboy/6267743
276 int int64_hash (void *data) {
277  uint64_t key = *(uint64_t *)data;
278  key = (~key) + (key << 21); // key = (key << 21) - key - 1;
279  key = key ^ (key >> 24);
280  key = (key + (key << 3)) + (key << 8); // key * 265
281  key = key ^ (key >> 14);
282  key = (key + (key << 2)) + (key << 4); // key * 21
283  key = key ^ (key >> 28);
284  key = key + (key << 31);
285  return (int)key;
286 }
