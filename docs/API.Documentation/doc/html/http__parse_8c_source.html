44 // qvalue = ( "0" [ "." 0*3DIGIT ] ) / ( "1" [ "." 0*3("0") ] )
45 char *qvalue (int *q, char *data) {
46  int n = 3, f = 0;
47  switch (*data++) {
48  case '0': *q = 0; break;
49  case '1': *q = 1000; break;
50  default: return NULL;

53  while (digit (*data) && n-- > 0)
54  f = f * 10 + (*data++ - '0');

56  if (*q == 1000 && f != 0) return NULL;

58  }

79 // parse media range/type
80 // "Accept" (RFC 7230 Appendix B. Collected ABNF)
81 // accepts more than one q parameter
82 char *media_range (MediaType *m, char *data) {
83  int state = 0, c, i = 0; char *param, *value;
84  ok (data);
85  memset (m->param, 0, 8 * sizeof (char *)); c = *data;
86  while (1) {
87  switch (state) {
88  case 0: switch (c) { // type
89  case '\0': return NULL;
90  case ' ': case '\t': case ',': data++; break;
91  default: ok (data = token (&m->type, data));
92  m->quality = 1000; state++; break;
93  } break;
94  case 1: ok (c == '/'); data++; state++; break;
95  case 2: // subtype
96  ok (data = token (&m->sub, data)); state++; break;
97  case 3: *data = '\0'; state++;
98  case 4: switch (c) { // parameter
99  case ',': data++;
100  case '\0': return data;
101  case ' ': case '\t': data++; break;
102  case ';': state++; data++; break;
103  default: return NULL;
104  } break;
105  case 5: // parameter name
106  ok (data = token (&param, ows (data))); state++; break;
107  case 6: ok (c == '='); // parameter value
108  *data = '\0'; to_lower (param);
109  if (streq (param, "q")) {
110  ok (data = qvalue (&m->quality, data+1)); state = 4;
111  } else {
112  ok (data = pvalue (&value, data+1)); state++;
113  } break;
114  case 7: *data = '\0'; // completed parameter
115  m->param[i] = param; m->param[i+1] = value; i = (i+2)&7;
116  state = 4; continue;
117  }
118  c = *data;
119  }
120 }
