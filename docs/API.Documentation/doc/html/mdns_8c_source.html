29 // parse a dns name - counted strings with pointers
30 // simple pointer loops are detected, complex ones are limited by name length
31 char *dns_name (char *name, char *data) {
32  char *ptr = NULL;
33  int n, name_length = 0, prev_length = -1;
34  while (!truncated (data+1)) {
35  if (*data == '\0') {
36  *name = '\0';
37  if (ptr) return ptr+2;
38  return data+1;
39  }
40  switch (*data & 0xc0) {
41  case 0: // label
42  n = *data + 1;
43  if ((name_length + n) > 254) return NULL;
44  if (truncated (data+n)) return NULL;
45  memcpy (name, data, n);
46  name += n; name_length += n; data += n;
47  break;
48  case 0xc0: // pointer
49  if (truncated (data+2)) return NULL;
50  if (ptr == NULL) ptr = data;
51  data = dns_start + (UNPACK16 (data) & 0x3fff);
52  if (truncated (data)) return NULL; // exceeds packet length
53  if (prev_length == name_length) return NULL; // pointer loop with no data
54  prev_length = name_length;
55  break;
56  default: printf ("dns_name: unknown type in name\n");
57  return NULL;
58  }
59  }
60  return NULL;
61 }

63 // encode dns name using PTR compression
64 char *encode_name (char *buffer, char *name, char *start, char **names) {
65  while (*name) { int i, n = name[0]+1;
66  for (i = 0; names[i] != NULL; i++) {
67  char *sub = names[i];
68  if (strncmp (name, sub, n) == 0) {
69  int ptr = 0xc000 | (sub - start);
70  PACK16 (buffer, ptr);
71  return buffer + 2;
72  }
73  }
74  names[i] = buffer;
75  memcpy (buffer, name, n);
76  name += n; buffer += n;
77  }
78  *buffer = '\0'; return buffer + 1;
79 }
