34 int mdns_response (UdpPort *mdns, DnsHeader *header,
35  char *data, int unicast) {
36  int i = 0, class, n, ancount = 0, qdcount = header->qdcount;
37  char *names[64] = {0}, response[DNS_PACKET_SIZE], *path[8];
38  int answer_ptr[8], ret = 0; const char *service;
39  char name[255], *packet = response+12, *end;
40  Address *address;
41  // process queries
42  while (qdcount--) {
43  if ((data = dns_name (name, data))
44  && !truncated (data+4)
45  && UNPACK16 (data) == DOMAIN_NAME_PTR
46  && (class = UNPACK16 (data+2))
47  && (class & 0x7fff) == INTERNET_CLASS) { data += 4;
48  if ((class & 0x8000) ^ unicast) { ret = 0x8000; continue; }
49  if ((service = dns_lookup (name, &path[i])) == NULL) continue;
50  packet = encode_name (packet, name, response, names);
51  packet = pack_record (packet, PTR_RECORD, 1, 5*60);
52  n = name[0] = sprintf (name+1, "%s-%012lld", service, device_sfdi);
53  n++; strcpy (name+n, sub_name+5);
54  end = encode_name (packet+2, name, response, names);
55  n = end - (packet+2); PACK16 (packet, n); // name length
56  answer_ptr[i] = 0xc000 | (packet + 2 - response);
57  packet = end; i++;
58  } else { format_error (mdns, header->id); return 0; }
59  } ancount = i; i = 0;
60  if (!ancount) return ret;
61  memset (response, 0, 12);
62  PACK16 (response, header->id);
63  PACK16 (response+2, 0x8400);
64  PACK16 (response+6, ancount);
65  PACK16 (response+10, ancount*2+1);
66  // additional records
67  while (ancount--) {
68  int answer = answer_ptr[i];
69  // SRV record
70  PACK16 (packet, answer);
71  packet = pack_record (packet+2, SRV_RECORD, 1, 5*60);
72  end = encode_name (packet+8, target_site, response, names);
73  n = end - (packet+2);
74  PACK16 (packet, n); // length
75  PACK16 (packet+2, 0); // priority
76  PACK16 (packet+4, 0); // weight
77  PACK16 (packet+6, 80); // port
78  packet = end;
79  // TXT record
80  PACK16 (packet, answer);
81  packet = pack_record (packet+2, TXT_RECORD, 1, 5*60);
82  n = sprintf (packet+2, ".txtvers=1.dcap=%s.level=-S1.https=443",
83  dcap_path);
84  if (path[i]) n += sprintf (packet+2+n, ".path=%s", path[i]);
85  write_counted (packet+2);
86  PACK16 (packet, n); // data length
87  packet += 2+n; i++;
88  }
89  // host record
90  packet = encode_name (packet, target_site, response, names);
91  packet = pack_record (packet, AAAA_RECORD, 1, 5*60);
92  PACK16 (packet, 16);
93  memcpy (packet+2, &server_address.in6.sin6_addr, 16);
94  address = unicast? &mdns->source : &multicast;
95  udp_send (mdns, response, packet+18-response, address);
96  return ret;
97 }

99 void mdns_serve (UdpPort *mdns, char *data, int length) {
100  DnsHeader header;
101  dns_start = data; dns_end = data + length;
102  if ((data = dns_header (&header, data))
103  && (header.flags & 0xfb70) == 0 // standard query
104  && header.qdcount > 0 // has queries
105  && header.ancount == 0
106  && header.nscount == 0
107  && header.arcount == 0) {
108  int unicast = mdns_response (mdns, &header, data, 0);
109  if (unicast) mdns_response (mdns, &header, data, unicast);
110  }
111 }
