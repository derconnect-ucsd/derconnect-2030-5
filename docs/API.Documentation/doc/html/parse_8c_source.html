186 void *parse_doc (Parser *p, int *type) {
187  const SchemaElement *se;
188  const ParserDriver *d = p->driver;
189  ElementStack *stack = &p->stack;
190  StackItem *t; int size;
191  while (1) {
192  switch (p->state) {
193  case PARSE_START:
194  ok (d->parse_start (p));
195  stack->n = 0; p->state++;
196  size = object_element_size (p->se, p->schema);
197  p->obj = p->base = calloc (1, size); break;
198  case PARSE_ELEMENT:
199  se = p->se; p->flag = se->bit;
200  if (se->attribute) {
201  if (!se->min && !is_pointer (se->xs_type)) {
202  set_count (p->base, 1, p->flag); p->flag++;
203  } p->state = PARSE_ATTRIBUTE; continue;
204  } else if (t = push_element (stack, se, p->base)) {
205  p->base += se->offset;
206  t->size = object_element_size (se, p->schema);
207  if (se->unbounded) {
208  List *l = *(List **)p->base = add_element (t);
209  p->base = l->data;
210  } else {
211  t->diff = se->max - se->min;
212  if (t->diff && !(se->simple && is_pointer (se->xs_type)))
213  p->flag += bit_count (t->diff);
214  }
215  } else goto parse_error;
216  parse_element:
217  if (se->simple) goto parse_value;
218  p->se = &p->schema->elements[se->index+1];
219  p->state = PARSE_NEXT;
220  case PARSE_NEXT:
221  ok (d->parse_next (p)); break;
222  case PARSE_ATTRIBUTE:
223  ok (d->parse_attr_value (p, p->base+p->se->offset));
224  p->state--; p->se++; break;
225  parse_value:
226  p->state = PARSE_VALUE;
227  case PARSE_VALUE:
228  ok (d->parse_value (p, p->base)); p->state++; break;
229  case PARSE_END:
230  if(stack->n) {
231  t = stack_top (stack); se = t->se;
232  ok (d->parse_end (p, se)); t->count++;
233  if (se->unbounded || t->count < se->max)
234  p->state = PARSE_SEQUENCE;
235  else p->state = SEQUENCE_END;
236  } else {
237  d->parse_done (p);
238  p->state = PARSE_START;
239  *type = p->type; return p->obj;
240  } break;
241  case PARSE_SEQUENCE:
242  t = stack_top (stack); se = t->se;
243  if (d->parse_sequence (p, t)) {
244  if (se->unbounded)
245  p->base = list_data (add_element (t));
246  else if (t->count < se->max)
247  p->base += t->size;
248  else goto parse_error;
249  p->flag++; goto parse_element;
250  } else if (p->state == PARSE_SEQUENCE)
251  return NULL; break;
252  case SEQUENCE_END:
253  if (t->diff)
254  set_count (t->base, t->count-se->min, se->bit);
255  p->se = pop_element (stack, &p->base)+1;
256  p->state = PARSE_NEXT; break;
257  case PARSE_INVALID: return NULL;
258  }
259  }
260  parse_error:
261  p->state = PARSE_INVALID; return NULL;
262 }
