7 int parse_query (Query *q, char *query) {
8  int state = 0, c, p; int64_t x;
9  if (!query) return 1;
10  q->start = 0; q->after = 0; q->limit = 1;
11  while (1) { c = *query;
12  switch (state) {
13  case 0: if (c == '\0') return 1; state++;
14  case 1: p = c; state++; break;
15  case 2: if (c != '=') return 0; state++; break;
16  case 3: if (digit (c)) x = (c - '0'); else return 0; state++; break;
17  case 4: switch (c) {
18  case '&': state = 0; query++;
19  case '\0': state++; switch (p) {
20  case 'a': q->after = x; break;
21  case 'l': q->limit = x; break;
22  case 's': q->start = x; break;
23  default: return 0;
24  } break;
25  default: if (digit (c)) x = x * 10 + (c - '0');
26  else return 0; query++;
27  } continue;
28  case 5: return 1;
29  }
30  query++;
31  }
32 }
