213 EventBlock *insert_event (EventBlock *head, Device *device, uint64_t now,
214  void *event, int type, int primacy) {
215  SE_Event_t *ev = event;
216  EventBlock *prev = NULL, *e, *t, tmp;
217  uint64_t start = ev->interval.start,
218  end = start + ev->interval.duration;
219  device_response (device, ev, EventReceived);
220  if (end <= now) { // specified end time is in the past
221  device_response (device, ev, EventExpired);
222  return head;
223  }
224  tmp.device = device;
225  tmp.primacy = primacy;
226  tmp.event = event;
227  tmp.type = type;
228  // skip events that end in the past
229  foreach (e, head) {
230  if (e->end > start) break; prev = e;
231  }
232  // handle past events that overlap
233  if (e && e->start < start) {
234  if (event_supersede (e, &tmp)) {
235  device_response (device, ev, EventSuperseded);
236  return head;
237  } else {
238  e->end = start; e->superseded = 1;
239  prev = e; e = e->next;
240  }
241  }
242  tmp.next = e;
243  // handle future events that overlap
244  while (e && e->start < end) {
245  if (event_supersede (e, &tmp)) {
246  tmp.superseded = 1;
247  end = e->start; break;
248  } else {
249  device_response (device, e->event, EventSuperseded);
250  t = e; e = e->next; free (t); continue;
251  } e = e->next;
252  }
253  // insert event if not truncated
254  if (start != end) {
255  EventBlock *eb = malloc (sizeof (EventBlock));
256  tmp.start = start; tmp.end = end;
257  memcpy (eb, &tmp, sizeof (EventBlock));
258  if (prev) prev->next = eb; else head = eb;
259  } else device_response (device, ev, EventSuperseded);
260  return head;
261 }
