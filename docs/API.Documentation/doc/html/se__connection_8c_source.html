187 int se_receive (void *conn) {
188  SeConnection *s = conn;
189  HttpConnection *h = conn;
190  Parser *p = &s->parser;
191  char *data, *m_name;
192  int length, type, code, method;
193  http_flush (h);
194  switch (method = http_receive (h)) {
195  case HTTP_NONE: break;
196  case HTTP_ERROR: return SE_ERROR;
197  default:
198  switch (s->state) {
199  case SE_START: p->obj = NULL;
200  print_http_status (h);
201  if (h->media_range)
202  s->media = select_media (h->media_range);
203  if (h->body) {
204  if (se_parse_init (s)) s->state++;
205  else { code = 415; goto error; }
206  } else return method;
207  case SE_DATA:
208  while (data = http_data (h, &length)) {
209  parser_rebuffer (p, data, length);
210  if (parse_doc (p, &type)) {
211  s->state = SE_START; return method;
212  } else if (!http_complete (h)) {
213  http_rebuffer (h, p->ptr);
214  } else {
215  printf ("parse error in message body\n");
216  print_parse_stack (p);
217  code = 400; goto error;
218  }
219  } set_timeout (s);
220  }
221  }
222  return SE_INCOMPLETE;
223  error:
224  if (h->method == HTTP_RESPONSE) http_close (h);
225  else http_error (h, code);
226  return SE_ERROR;
227 }
