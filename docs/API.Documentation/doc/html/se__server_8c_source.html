32 void se_serve (void *conn) {
33  switch (conn_session (conn)) {
34  case SESSION_NEW: check_session (conn);
35  case SESSION_CONNECTED:
36  switch (se_receive (conn)) {
37  case HTTP_GET: get_resource (conn, HTTP_GET); break;
38  case HTTP_HEAD: get_resource (conn, HTTP_HEAD); break;
39  case HTTP_POST:
40  if (!event_response (conn))
41  post_resource (conn); break;
42  case HTTP_PUT: put_resource (conn); break;
43  case HTTP_UNKNOWN: http_allow (conn, "GET, HEAD, POST, PUT, DELETE");
44  default: return;
45  }
