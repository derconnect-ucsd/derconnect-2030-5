64 uint64_t sfdi_gen(uint8_t *lfdi) {
65  int i = 0; uint64_t sfdi = 0;
66  while (i < 5) sfdi = (sfdi << 8) + lfdi[i++];
67  sfdi >>= 4; sfdi = sfdi * 10 + check_digit (sfdi);
68  return sfdi;
69 }
