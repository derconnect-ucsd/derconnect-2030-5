72  char *data = (char *)p->ptr;
73  int n = type >> 4;
74  switch (type & 0xf) {
75  case XS_STRING: if (n) {
76  if (strlen (data) > n-1) return 0;
77  strcpy (value, data);
78  } else *(char **)(value) = strdup (data); return 1;
79  case XS_BOOLEAN: if (streq (data, "true") || streq (data, "1"))
80  *(uint32_t *)value |= 1 << p->flag;
81  else if (!(streq (data, "false") || streq (data, "0"))) {
82  p->state = PARSE_INVALID; break; } return 1;
83  case XS_HEX_BINARY: return parse_hex (value, n, data);
84  case XS_ANY_URI: *(char **)(value) = strdup (data); return 1;
