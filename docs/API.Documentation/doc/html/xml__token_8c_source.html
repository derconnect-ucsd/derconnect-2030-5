29 int name_start (int c) {
30  return alpha (c) || c == ':' || c == '_'
31  || in_range (c, 0xc0, 0xd6) || in_range (c, 0xd8, 0xf6)
32  || in_range (c, 0xf8, 0x2ff) || in_range (c, 0x370, 0x37d)
33  || in_range (c, 0x37f, 0x1fff) || in_range (c, 0x200c, 0x200d)
34  || in_range (c, 0x2070, 0x218f) || in_range (c, 0x2c00, 0x2fef)
35  || in_range (c, 0x3001, 0xd7ff) || in_range (c, 0xf900, 0xfdcf)
36  || in_range (c, 0xfdf0, 0xfffd) || in_range (c, 0x10000, 0xeffff);

68 char *hex_number (int *x, char *data) {
69  int c, d, y = 0, n = 0;
70  while (c = hex_digit (d = *data))
71  y = (y << 4) | (d - c), n++, data++;
72  return n > 0? *x = y, data : NULL;

75 char *xml_reference (char **ptr, char *data) {
76  const char * const refs[] = {"amp", "lt", "gt", "quot", "apos"};
77  char *end = strchr (data, ';'); int ref = 0;
78  if (!end) return NULL; *end++ = '\0';
79  if (*data == '#') { data++;
80  data = *data == 'x'? hex_number (&ref, data+1)
81  : number (&ref, data);
82  return data && *data == '\0'?
83  *ptr = utf8_encode (*ptr, ref), end : NULL;
84  }
85  switch (string_index (data, refs, 5)) {
86  case 0: ref = '&'; break;
87  case 1: ref = '<'; break;
88  case 2: ref = '>'; break;
89  case 3: ref = '\"'; break;
90  case 4: ref = '\''; break;
91  default: return NULL;
92  } *(*ptr)++ = ref; return end;
93 }

118 char *xml_attributes (char **attr, char *data) {
119  char *next, *end; int i = 0; data = trim (data);
120  memset (attr, 0, sizeof(char *)*MAX_ATTRIBUTE*2);
121  while (end = xml_name (data)) {
122  char *value = xml_eq (end);
123  if (value && (next = att_value (value++))) {
124  *end = '\0'; attr[i] = data; attr[i+1] = value;
125  i = (i + 2) % (2 * MAX_ATTRIBUTE);
126  } else return NULL;
127  data = trim (next);
128  } return data;

143 char *xml_tag (XmlParser *p, char *data) { char *end;
144  p->token = *data == '/'? data++, END_TAG : START_TAG;
145  p->name = data; ok (data = end = xml_name (data));
146  if (p->token == START_TAG) {
147  if (ws (*data))
148  ok (data = xml_attributes (p->attr, data+1));
149  if (*data == '/') data++, p->token = EMPTY_TAG;
150  } else data = trim (data);
151  ok (*data == '>'); *end = '\0';
152  return data+1;

155 // scan an xml token, return the token type
156 int xml_token (XmlParser *p) {
157  int state = p->state, c;
158  char *data = p->data, *next, *text;
159  if (p->token == XML_TEXT) text = p->content+p->length;
160  while (c = *data) { next = data+1;
161  switch (state) {
162  case 0: // initial state
163  if (c == '<') state = 2;
164  else if (ws (c)) next = trim (next);
165  else {
166  p->content = text = data;
167  p->token = XML_TEXT;
168  state++; continue;
169  } break;
170  case 1: // text
171  switch (c) {
172  case '&':
173  if (strchr (next, ';')) {
174  if (!(next = xml_reference (&text, next)))
175  return XML_INVALID;
176  } else goto incomplete; break;
177  case '<': state++; break;
178  default:
179  if (next = utf8_char (&c, data)) {
180  if (xml_char (c))
181  text = utf8_encode (text, c);
182  else return XML_INVALID;
183  } else goto incomplete;
184  } break;
185  case 2: // "<" (tag)
186  switch (c) {
187  case '!': state = 4; break;
188  default:
189  if (p->token == XML_TEXT) {
190  p->state = 2; p->data = data;
191  *text = '\0';
192  p->length = text - p->content;
193  p->token = XML_NONE; return XML_TEXT;
194  }
195  if (c == '?') { state++; break; }
196  if (strchr (next, '>'))
197  return (p->data = xml_tag (p, data))?
198  p->state = 0, p->token : XML_INVALID;
199  goto incomplete;
200  } break;
201  case 3: // "<?" (processing instruction)
202  if (next = token_end (data, "?>", 2))
203  return xml_pi (p, data)? p->data = next, p->token : XML_INVALID;
204  goto incomplete;
205  case 4: // "<!" (comment, CDATA)
206  switch (c) {
207  case '-': state++; break;
208  case '[':
209  if ((next = token_end (next, "]]>", 3))) {
210  if (streq (data+1, "CDATA[")) {
211  int n = (next-3) - (data+7);
212  memmove (text, data+7, n);
213  text += n; state = 1;
214  } else return XML_INVALID;
215  } else goto incomplete; break;
216  default: return XML_INVALID;
217  } break;
218  case 5: // "<!-" (comment)
219  if (c == '-') state++;
220  else return XML_INVALID;
221  break;
222  case 6: // "<!--" (comment)
223  if ((next = token_end (data, "-->", 3))) {
224  if (*(next-4) == '-') return XML_INVALID;
225  state = p->token == XML_TEXT? 1 : 0; break;
226  } goto incomplete;
227  } data = next;
228  }
229  incomplete:
230  if (p->token == XML_TEXT) p->length = text - p->content;
231  else p->content = data;
232  p->state = state; p->data = data;
233  return XML_INCOMPLETE;
234 }
