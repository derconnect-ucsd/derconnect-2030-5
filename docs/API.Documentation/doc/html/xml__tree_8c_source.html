107 Element *xml_parse (char *data) {
108  XmlParser p; Element *e, root; char *prefix, *name;
109  memset (&root, 0, sizeof(Element)); e = &root;
110  xml_init (&p, data);
111  while (1) {
112  switch (xml_token (&p)) {
113  case START_TAG: e = new_element (e, p.name, p.attr); break;
114  case EMPTY_TAG: new_element (e, p.name, p.attr); break;
115  case END_TAG:
116  name = local_name (&prefix, p.name);
117  if (!e->parent || strcmp (name, e->name) != 0
118  || strcmp (prefix, e->prefix) != 0) {
119  free_node (root.first);
120  return NULL;
121  }
122  e = e->parent;
123  if (!e->parent) {
124  return root.first;
125  }
126  break;
127  case XML_DECL: break;
128  case XML_PI: new_content (e, PI_NODE, p.content); break;
129  case XML_TEXT: new_text (e, TEXT_NODE, p.content, p.length); break;
130  default: return NULL;
131  }
132  }
133 }
